3. Programming Sample

A PHP function demonstrating the manipulation of associative arrays to generate dynamic HTML content.

<?php
/**
 * Generates HTML links from an associative array.
 * Demonstrates: Loops, Arrays, String Interpolation.
 */
function makeLinks($array) {
    foreach ($array as $name => $url) {
        // Safe output with clean HTML structure
        echo "<a href='$url' target='_blank' class='nav-link'>$name</a><br>";
    }
}

// Data Source
$links = array(
    "Outlook Mail" => "https://outlook.com",
    "Student Portal" => "https://lethpolytech.ca",
    "Database Tools" => "https://dbeaver.io"
);

// Execution
makeLinks($links);
?>
