CREATE PROCEDURE sp_CalculateShiftEarnings(
    IN emp_id_param INT,
    IN shift_date DATE
)
BEGIN
    -- Calculate total sales and tips for specific shift
    SELECT 
        e.employee_name,
        COUNT(t.transaction_id) AS total_appointments,
        SUM(t.tip_amount) AS total_tips,
        CASE 
            WHEN SUM(t.tip_amount) > 100 THEN 'High Performance'
            ELSE 'Standard'
        END AS performance_rating
    FROM employees e
    JOIN transactions t ON e.id = t.emp_id
    WHERE e.id = emp_id_param 
      AND DATE(t.created_at) = shift_date
    GROUP BY e.id;
END //
